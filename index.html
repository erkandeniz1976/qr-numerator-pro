let AKTIF_ID = "000"; // Sayfa yüklendiğinde çalışacak ana başlatıcı fonksiyon function sayfaBaslat() { const urlParams = new URLSearchParams(window.location.search); AKTIF_ID = urlParams.get('id'); // Eğer parametresiz ana site linki açılırsa otomatik Fabrika Moduna geç if (!AKTIF_ID) { document.getElementById('id-rozeti').innerText = "FABRİKA MODU"; document.getElementById('main-title').innerText = "QR Üretim Fabrikası"; document.getElementById('main-message').innerText = "Matbaa basımı için toplu QR kodu üretebilirsiniz."; ekranDegistir('qr-uretim-fabrikasi'); return; } // ID varsa normal etiket modunda başlat document.getElementById('id-rozeti').innerText = "ID: #" + AKTIF_ID; // Yerel depolamada bu ID kayıtlı mı kontrol et (Simüle edilmiş veritabanı) let etiketVerisi = localStorage.getItem("qr_" + AKTIF_ID); if (!etiketVerisi) { // Etiket daha önce kurulmamışsa aktivasyon ekranına yönlendir document.getElementById('main-message').innerText = "Kurulmamış Etiket"; ekranDegistir('aktivasyon-ekrani'); } else { // Etiket aktifse vatandaş doğrudan durumu görür document.getElementById('main-message').innerText = "Aktif Akıllı QR Etiketi"; vatandasEkraniniAc(); } } // Ekranlar arası geçişi sağlayan yardımcı fonksiyon function ekranDegistir(hedefEkranId) { const ekranlar = ['aktivasyon-ekrani', 'login-ekrani', 'panel-ekrani', 'vatandas-ekrani', 'qr-uretim-fabrikasi']; ekranlar.forEach(id => { document.getElementById(id).classList.add('hidden'); }); document.getElementById(hedefEkranId).classList.remove('hidden'); } // 0. ADIM: ETIKETI AKTIF ETME function etiketAktifEt() { let kanal = document.getElementById('act-channel').value.trim(); let pin = document.getElementById('act-pin').value.trim(); if (!kanal || pin.length < 4) { alert("Lütfen geçerli bir gizli kanal adı yazın ve 4 haneli PIN belirleyin!"); return; } let yeniEtiket = { kanal: kanal, pin: pin, aktifMod: "🚗 Hatalı Park Modu" // Varsayılan ilk mod }; localStorage.setItem("qr_" + AKTIF_ID, JSON.stringify(yeniEtiket)); alert("🚀 Etiket başarıyla aktifleştirildi! Yönetim paneline aktarılıyorsunuz."); ekranDegistir('panel-ekrani'); } // 1. ADIM: SÜRÜCÜ GİRİŞİ DOĞRULAMA function paneleGeriGirisYap() { let girilenPin = document.getElementById('login-pin').value; let etiketVerisi = JSON.parse(localStorage.getItem("qr_" + AKTIF_ID)); if (etiketVerisi && etiketVerisi.pin === girilenPin) { document.getElementById('login-pin').value = ""; // Temizle ekranDegistir('panel-ekrani'); } else { alert("❌ Hatalı PIN kodu girdiniz! Lütfen tekrar deneyin."); } } // 2. ADIM: MOD SEÇİMİ VE ARAMA FİLTRESİ function modSec(secilenMod) { let etiketVerisi = JSON.parse(localStorage.getItem("qr_" + AKTIF_ID)); if (etiketVerisi) { etiketVerisi.aktifMod = secilenMod; localStorage.setItem("qr_" + AKTIF_ID, JSON.stringify(etiketVerisi)); alert("Durum başarıyla güncellendi: " + secilenMod); } } // Modları anlık arama/filtreleme fonksiyonu function modFiltrele() { let input = document.getElementById('mod-search').value.toLowerCase(); let butonlar = document.getElementsByClassName('mod-btn'); let eşleşmeVar = false; for (let i = 0; i < butonlar.length; i++) { let modMetni = butonlar[i].innerText.toLowerCase(); if (modMetni.includes(input)) { butonlar[i].classList.remove('hidden'); eşleşmeVar = true; } else { butonlar[i].classList.add('hidden'); } } // Hiçbir mod bulunamazsa hata mesajı göster let hataMesaji = document.getElementById('hata-mesaji'); if (eşleşmeVar) { hataMesaji.classList.add('hidden'); } else { hataMesaji.classList.remove('hidden'); } } // 3. ADIM: VATANDAŞ EKRANI GÖRÜNÜMÜ function vatandasEkraniniAc() { let etiketVerisi = JSON.parse(localStorage.getItem("qr_" + AKTIF_ID)); let mesajAlani = document.getElementById('vatandas-mesaji'); if (etiketVerisi) { mesajAlani.innerText = "Sürücü Durumu: " + etiketVerisi.aktifMod; } else { mesajAlani.innerText = "Sürücü Durumu: 🚗 Hatalı Park Modu (Önizleme)"; } ekranDegistir('vatandas-ekrani'); } function paneleGeriDon() { ekranDegistir('login-ekrani'); } function cikisYap() { ekranDegistir('login-ekrani'); } // 4. ADIM: MATBAA İÇİN TOPLU QR KODU ÜRETİM FABRİKASI function topluQrKoduUret() { let startId = parseInt(document.getElementById('start-id').value); let endId = parseInt(document.getElementById('end-id').value); let baskiAlani = document.getElementById('qr-baski-alani'); if (isNaN(startId) || isNaN(endId) || startId > endId) { alert("Lütfen geçerli bir başlangıç ve bitiş ID aralığı girin!"); return; } baskiAlani.innerHTML = ""; // Eski üretimleri temizle // Belirtilen aralıkta döngü kurarak 5x5 cm şablonları çoğalt for (let id = startId; id <= endId; id++) { // Mevcut sayfanın URL'ini temel alarak her ID için benzersiz link oluşturur let targetUrl = window.location.origin + window.location.pathname + "?id=" + id; // Güvenli Google Charts API kullanarak dinamik QR kodu oluşturma linki let qrImgSrc = "https://googleapis.com" + encodeURIComponent(targetUrl) + "&choe=UTF-8"; // Matbaa şablonu HTML yapısı let sablonHtml = `
AKILLI QR ETİKETİ
SERİ NO: #${id}
Kameranızla taratıp sürücüye ulaşın.
`; baskiAlani.innerHTML += sablonHtml; } } // Sayfa yüklendiğinde kurulumu otomatik tetikle window.onload = sayfaBaslat;
